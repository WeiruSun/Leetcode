 ```
 class Solution:
    def canAttendMeetings(self, intervals: List[List[int]]) -> bool:
        
        #overlapping [[1,2][2,3]]
        
        #sort intervals
        intervals.sort(key= lambda i : i[0])

        for i in range(1, len(intervals)):
            i1 = intervals[i-1]
            i2 = intervals[i]

            #check overlapping
            if i1[1] > i2[0]:
                return False
   
        return True
        
        #time complexity : O(nlog(n))
```
