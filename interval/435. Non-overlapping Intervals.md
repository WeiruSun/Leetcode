```Python
class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        res = 0

        #sort
        intervals.sort(key=lambda i : i[0])
        
        tail = intervals[0][1]

        for start,end in intervals[1:]:
            # not overlapping
            if start >= tail:
                tail = max(tail,end)
            # overlapping
            else:
                tail = min(tail,end)
                res +=1
        
        return res
        # time complexity: O(log(n))
```
