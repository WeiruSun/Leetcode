 ```Python
 class Solution:
    def validPalindrome(self, s: str) -> bool:
        deleteChar = 1
        left,right = 0, len(s)-1
        while left < right:
            if s[left] != s[right]:
                if deleteChar !=0:
                    deleteChar -=1
                else:
                    return False
            left +=1
            right -=1
            
        return True

```
